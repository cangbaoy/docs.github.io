# RobotRecordMaintenanceHistory

package com.swagger.maintenance.entity;

import lombok.Data;

import javax.persistence.Column;
import javax.persistence.Id;
import javax.persistence.Table;
import java.io.Serializable;
import java.util.Date;
/**
 * 钉钉成功历史 Entity
 * <p>Title: RobotRecordMaintenanceHistory</p>
 * 
 * @author penghe.yu
 * @date 2019-08-02 21:54
 * @package com.swagger.maintenance.entity
 */
@Data
@Table(name = "tb_dingding_maintenance_history")
public class RobotRecordMaintenanceHistory implements Serializable {
    @Id
    @Column(name = "id")
    private Long id;

    /**
     * 通知内容
     */
    @Column(name = "content")
    private String content;

    /**
     * 消息类型(0:text，1：链接，2：markdown)
     */
    @Column(name = "send_type")
    private String sendType;

    /**
     * 消息发送时间
     */
    @Column(name = "send_time")
    private Date sendTime;

    /**
     * 钉钉客户端（群机器人ID）
     */
    @Column(name = "client_id")
    private String clientId;

    /**
     * 备注
     */
    @Column(name = "remarks")
    private String remarks;

    /**
     * @ 的群成员（手机号，分隔）
     */
    @Column(name = "mobiles")
    private String mobiles;

    /**
     * 链接消息url
     */
    @Column(name = "link_url")
    private String linkUrl;

    /**
     * 发送人
     */
    @Column(name = "user_id")
    private String userId;

    /**
     * 平台flag(0：wish平台, 1：ebay, 2： 新平台 3: BI)
     */
    @Column(name = "platform_flag")
    private String platformFlag;

    /**
     * 状态(0: 待处理，1: 处理中 , 2: 异常, 3: 成功)
     */
    @Column(name = "`status`")
    private String status;

    /**
     * 创建人
     */
    @Column(name = "create_id")
    private String createId;

    /**
     * 创建时间
     */
    @Column(name = "create_time")
    private Date createTime;

    /**
     * 修改时间
     */
    @Column(name = "update_time")
    private Date updateTime;

    private static final long serialVersionUID = 1L;

    public RobotRecordMaintenanceHistory() {
        super();
    }

    public RobotRecordMaintenanceHistory(Long id, String content, String sendType, Date sendTime, String clientId,
                                         String remarks, String mobiles, String linkUrl, String userId,
                                         String platformFlag, String status, String createId, Date createTime,
                                         Date updateTime) {
        super();
        this.id = id;
        this.content = content;
        this.sendType = sendType;
        this.sendTime = sendTime;
        this.clientId = clientId;
        this.remarks = remarks;
        this.mobiles = mobiles;
        this.linkUrl = linkUrl;
        this.userId = userId;
        this.platformFlag = platformFlag;
        this.status = status;
        this.createId = createId;
        this.createTime = createTime;
        this.updateTime = updateTime;
    }
}
